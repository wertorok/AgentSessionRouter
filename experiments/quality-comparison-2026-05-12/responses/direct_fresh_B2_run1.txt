### 4. `CliClaudeAdapter.sessionFileExists` (`src/claude.ts:104–122`)

```typescript
async sessionFileExists(claudeSessionId: string): Promise<boolean> {
  const candidates = [
    path.join(home, ".claude", "sessions", `${claudeSessionId}.json`),
    path.join(home, ".claude", "sessions", `${claudeSessionId}.jsonl`),
    path.join(home, ".claude", "projects")
  ];

  if (existsSync(candidates[0]) || existsSync(candidates[1])) {
    return true;
  }

  // Also searches ~/.claude/projects/ recursively up to depth 3
  return findFile(projectsDir, `${claudeSessionId}.jsonl`, 3);
}
```

**If the `.jsonl` is deleted**, both `.json` and `.jsonl` checks return `false`, and `findFile` over `~/.claude/projects/` also finds nothing → returns `false`.
